üßÆ Step 4: Create an Interactive Program ‚Äì Multiply

dotnet new console -n multiply
cd multiply

Edit Program.cs:

using System;

class Program
{
    static void Main()
    {
        // Declare variables
        string response;
        double number1, number2;
        double product;

        // Input
        Console.Write("Enter the first number: ");
        response = Console.ReadLine();
        number1 = Convert.ToDouble(response);

        Console.Write("Enter the second number: ");
        response = Console.ReadLine();
        number2 = Convert.ToDouble(response);

        // Process
        product = number1 * number2;

        // Output
        Console.WriteLine("The product is: " + product);
    }
}

Run it:

dotnet run

üöó Step 5: Add Another Program ‚Äì kmToMiles

cd ..
dotnet new console -n kmToMiles
cd kmToMiles

Edit Program.cs:

using System;

class Program
{
    static void Main()
    {
        const double MILES_PER_KM = 0.62;

        Console.Write("Enter distance in kilometers: ");
        string input = Console.ReadLine();
        double km = Convert.ToDouble(input);
        double miles = km * MILES_PER_KM;

        Console.WriteLine($"{km} kilometers is about {miles} miles.");
    }
}

Run it:

dotnet run
